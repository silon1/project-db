SELECT b.building_id
FROM Building b
WHERE ESIXT(SELECT *
			FROM Room r
			WHERE r.department_id = 20
				  and r.room_type_id = 5
				  and r.building_id = b.building_id)

select b.building_id 
from Buildings b
where b.building_id in ( 
        select x.building_id
        from ( select r.building_id, count(r.room_id) c
               from rooms r
               where r.room_type_id = 1
               group by r.building_id ) x
        where x.c > 7)
