# כל מה שצריך להזמין
SELECT e.equipment_name e.min_amount - e.amount
FROM Equipment e
WHERE e.amount < e.min_amount

SELECT b.building_id
FROM Building b
WHERE ESIXT(SELECT *
			FROM Room r
			WHERE r.department_id = 20
				  and r.room_type_id = 5
				  and r.building_id = b.building_id)

# כל הביינים שיש בהם שואו ואין מנהלה
SELECT b1.building_id
FROM Buildings b1
WHERE EXISTS( SELECT *
            FROM Rooms r
            WHERE r.room_type_id = 1
            and r.building_id = b1.building_id)
INTERSECT
SELECT b2.building_id
FROM Buildings b2
WHERE NOT EXISTS( SELECT *
                FROM Rooms r
                WHERE r.room_type_id = 5
                and r.building_id = b2.building_id)
